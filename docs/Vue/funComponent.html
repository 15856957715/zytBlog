xianghaifeng 2021-03-11

# Vue2

// Vue 2 函数式组件示例
export default {
  functional: true,
  props: ['level'],
  render(h, { props, data, children }) {
    return h(`h${props.level}`, data, children)
  }
}

}

DynamicHeading.props = ['level']

export default DynamicHeading
